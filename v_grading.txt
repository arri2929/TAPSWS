SELECT A.NO_TICKET,A.TAHUN_TANAM,A.JML_JJG,JML_JJG_GRADING,A.INPUT_DATE,
B.SORTASI_CODE,C.GD_DESC,B.JML,B.BERAT,B.PERSEN ,B.REMARKS
FROM t_sortasi_header A
JOIN t_sortasi_detail B ON b.no_ticket=a.no_ticket
LEFT JOIN t_lookup_group_detail C ON GROUP_ID=5 AND TRIM(B.SORTASI_CODE)=SUBSTR(TRIM(C.GD_ID),2)
GROUP BY A.NO_TICKET,A.TAHUN_TANAM,A.JML_JJG,JML_JJG_GRADING,A.INPUT_DATE,SORTASI_CODE,C.GD_ID,C.GD_DESC
,B.SORTASI_CODE,B.JML,B.BERAT,B.PERSEN ,B.REMARKS
ORDER BY A.NO_TICKET,SORTASI_CODE
